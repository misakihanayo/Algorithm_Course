\subsection{Prob1}
\begin{algorithm}
    \begin{algorithmic}[1]
    \Procedure{Makeset}{$*A,v$}
        \State A->key=v;
        \State A->parent=A;
    \EndProcedure
    \end{algorithmic}
\end{algorithm}
\begin{algorithm}
    \begin{algorithmic}
        \Procedure{Findset}{$*A$}
            \If{A->parent->key==A->key}
                \State $return A$
            \Else
                \State $A->parent\gets Findset(A->parent)$
                \State $return A$
            \EndIf
        \EndProcedure
    \end{algorithmic}
\end{algorithm}
\begin{algorithm}
    \begin{algorithmic}
        \Procedure{Union}{$*A,*B$}
            \State $Aset=Findset(A)$
            \State $Bset=Findset(B)$
            \State $Bset->parent=Aset$
        \EndProcedure
    \end{algorithmic}
\end{algorithm}
